document.addEventListener('DOMContentLoaded', function() { // 页面淡入效果 document.body.style.opacity = 0; document.body.style.transition = 'opacity 0.6s ease'; setTimeout(() => { document.body.style.opacity = 1; }, 80); // 打字机口号 const slogan = document.getElementById('typewriter'); if (slogan) { const text = slogan.innerText.trim(); slogan.innerText = ''; let i = 0; const interval = setInterval(() => { if (i < text.length) { slogan.innerText += text.charAt(i); i++; } else { clearInterval(interval); } }, 80); } });
